\documentclass{standalone}
\usepackage[american,siunitx]{circuitikz}

\begin{document}

% First Circuit
\begin{circuitikz}
\draw
  (0,0) to[V, V<=15V, invert] (0,2) -- (1,2)
  to[R, l=$R_x$] (3,2)
  to[R, l=$R_y=5\Omega$] (3,0) -- (0,0)
  (3,2) -- (5,2)
  to[V, V<=2V] (5,0) -- (3,0);
\end{circuitikz}

% Second Circuit (Rx components)
\begin{circuitikz}

  (2,0) -- (4,0)
